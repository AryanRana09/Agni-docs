---
title: "Create Agent"
openapi: "POST /api/v1/agents"
description: "Create a new agent."
---

<ParamField header="Authorization" type="string" required>
  Bearer token in format `Bearer YOUR_API_KEY`.
</ParamField>

### Body

<ParamField body="organization_id" type="string" required>
  The unique identifier of the organization.
</ParamField>

<ParamField body="name" type="string" required>
  Internal name for the agent.
</ParamField>

<ParamField body="agent_name" type="string" required>
  The name of the agent that will be shown or spoken to users.
</ParamField>

<ParamField body="model" type="string" required>
  The language model to use (e.g., `gpt-4`).
</ParamField>

<ParamField body="temperature" type="number">
  Controls randomness in the response. Higher values mean more random responses.
</ParamField>

<ParamField body="begin_message" type="string">
  The first message the agent will speak when the call connects.
</ParamField>

<ParamField body="start_speaker" type="string">
  Who should speak first: `agent` or `user`.
</ParamField>

<ParamField body="ivr_options" type="object">
  Configuration options for IVR (Interactive Voice Response).
</ParamField>

<ParamField body="selected_tools" type="object">
  Tools enabled for the agent.
</ParamField>

<ParamField body="integrations" type="object">
  Integration settings for the agent.
</ParamField>

<RequestExample>

```json
{
  "organization_id": "org_123",
  "name": "Support Agent",
  "agent_name": "Maya",
  "model": "gpt-4",
  "temperature": 0.7,
  "begin_message": "Hello how can I help?",
  "start_speaker": "agent",
  "ivr_options": {},
  "selected_tools": {},
  "integrations": {}
}
```

</RequestExample>

### Response

<ResponseField name="success" type="boolean" required>
  Indicates if the request was successful.
</ResponseField>

<ResponseField name="message" type="string" required>
  A message indicating the result of the operation.
</ResponseField>

<ResponseField name="data" type="object" required>
  The created agent data.
  <Expandable title="properties">
    <ResponseField name="id" type="integer" required>
      The unique identifier of the new agent.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseExample>

```json 200
{
  "success": true,
  "message": "Agent created",
  "data": {
    "id": 1
  }
}
```

</ResponseExample>
