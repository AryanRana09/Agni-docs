---
title: "Create widget settings"
api: "POST /api/v1/widget-settings"
description: "Create a new widget settings configuration for deploying web agents."
---

### Authorizations

<ParamField header="Authorization" type="string" required>
  Authentication header containing API token (find it in dashboard). The format is "Bearer YOUR_API_KEY".
</ParamField>

<ParamField header="Content-Type" type="string" required>
  Must be set to `application/json`.
</ParamField>


### Body

<ParamField body="bot_name" type="string" required>
  The display name for the bot shown in the widget interface.
  
  **Example:**
  `"Ava"`
</ParamField>

<ParamField body="bot_auto_start" type="boolean" required>
  Whether the widget automatically initiates the conversation when loaded.
  
  **Example:**
  `true`
</ParamField>

<ParamField body="bot_position" type="string" required>
  The on-screen position where the widget will be rendered.
  
  **Available options:** `BOT_POSITION_BOTTOM_RIGHT`, `BOT_POSITION_BOTTOM_LEFT`, `BOT_POSITION_TOP_RIGHT`, `BOT_POSITION_TOP_LEFT`
  
  **Example:**
  `"BOT_POSITION_BOTTOM_RIGHT"`
</ParamField>

<ParamField body="bot_logo" type="string">
  Direct URL to an image asset (PNG, JPG, SVG) representing the bot's avatar.
  
  **Example:**
  `"https://cdn.ravan.ai/bot-logo.svg"`
</ParamField>

<ParamField body="svg_logo" type="string">
  Inline SVG code to be rendered directly as the bot logo.
</ParamField>

<ParamField body="custom_form_fields" type="object" required>
  JSON schema mapping custom fields you want to collect from the user before starting a chat.
  
  <Expandable title="Field Item Attributes">
    <ParamField body="[field_name].type" type="string">
      Type of the field (e.g., text, email).
    </ParamField>
    <ParamField body="[field_name].required" type="boolean">
      Whether completing this field is mandatory.
    </ParamField>
  </Expandable>

  **Example:**
  ```json
  {
    "company": {
      "type": "text",
      "required": true
    }
  }
  ```
</ParamField>

<ParamField body="theme" type="object" required>
  Configuration object to customize widget colors and styles.
  
  <Expandable title="Show child attributes">
    <ParamField body="primary" type="string">
      Primary hex color code.
    </ParamField>
  </Expandable>

  **Example:**
  ```json
  {
    "primary": "#16A34A"
  }
  ```
</ParamField>

<ParamField body="language" type="string" required>
  The localization language code for the widget interface.
  
  **Example:**
  `"en"`
</ParamField>

<ParamField body="consent_content" type="string">
  The explicit text content shown to the user asking for recording or interaction consent.
  
  **Example:**
  `"By continuing, you agree to recording."`
</ParamField>

<ParamField body="consent_title" type="string">
  The headline text for the consent dialog.
  
  **Example:**
  `"Consent"`
</ParamField>

### Response

<ResponseField name="success" type="boolean" required>
  Whether the request succeeded.
  
  **Example:**
  `true`
</ResponseField>

<ResponseField name="message" type="string" required>
  Response message indicating the status of the operation.
  
  **Example:**
  `"Widget settings created successfully"`
</ResponseField>

<ResponseField name="data" type="object">
  The generated widget settings configuration.
  
  <Expandable title="Show child attributes">
    <ResponseField name="id" type="integer" required>
      Widget settings ID.
    </ResponseField>
    
    <ResponseField name="bot_name" type="string" required>

    <ResponseField name="bot_auto_start" type="boolean" required>

    <ResponseField name="bot_logo" type="string">

    </ResponseField>

    <ResponseField name="custom_form_fields" type="object" required>

    </ResponseField>

    </ResponseField>

    <ResponseField name="consent_content" type="string">

    </ResponseField>

    <ResponseField name="consent_title" type="string">
